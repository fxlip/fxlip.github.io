<script>
  /* KERNEL 404: CAPTURA DE ROTA 
     Injeta a URL errada no terminal para realismo.
  */
  (function() {
    const currentPath = window.location.pathname;
    
    // Elementos Alvo (Agora únicos)
    const cmdElement = document.getElementById('cmd-display');
    const errElement = document.getElementById('err-display');
    
    // Trunca URLs gigantes para não quebrar o layout mobile
    const displayPath = currentPath.length > 30 
      ? currentPath.substring(0, 27) + "..." 
      : currentPath;

    // 1. Atualiza o comando digitado
    if(cmdElement) {
      cmdElement.innerText = "cd " + displayPath;
    }
    
    // 2. Atualiza a mensagem de erro (Opcional, mas mantido para consistência)
    if(errElement) {
      // Reconstrói a mensagem mantendo o link funcional
      errElement.innerHTML = `bash: cd: ${displayPath}: Arquivo ou diretório inexistente.<br>Voltar pro <a href='/' class='mention-link'>@feed</a>?`;
    }
  })();
</script>
